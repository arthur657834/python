def addHttpCode(statuscode, code_list):
    if statuscode < 300:
        code_list[0] += 1
    elif 300 <= statuscode < 400:
        code_list[1] += 1
    elif 400 <= statuscode < 500:
        code_list[2] += 1
    elif statuscode >= 500:
        code_list[3] += 1


def sumHttpCode(StatusCode, sumStatusCode):
    for i in xrange(4):
        sumStatusCode[i] += StatusCode[i]
